; ******** wall and tower animations
; ******** Kill zones go after duration line in transition scripts.  You can have 
; ******** as many as you like.  Defined as 'kill_zone <tick> <cas>',
; ******** eg 'kill_zone 20 my_kill_zone.cas', root directory is models_building/kill_zone.
; Params for dust_anim command:
; 	<start_tick> dust_anim <duration_ticks> <size> <quantity> <frequency> <start_anim_time> <end_anim_time>
; Starting from start_tick, until start_tick + duration_ticks, every 'frequency' ticks this will add
; 'quantity' dust clouds of radius 'size', at positions determined by the positions of geometries at the
; appropriate time in the animation.  It picks geometries at random, and the time in the animation is
; defined separately - ie it doesn't have to be the current positions, so you can create dust late or 
; early if it looks wrong.
; Wall damage:  collapse in five seconds, hold for two, then fade out over five.

	chiv_castle_wall_straight
	{
		full_health 2000
		battle_stats flammability_none impact_damage_low
		localised_name wall
		level
		{
			min_health 1000
			battle_stats flammability_none impact_damage_low
			item		chiv_castle_wall_straight
			physical_info	info_walls/info_roman_town_wall_straight.cas
		}
		transition ; undamaged to small damage
		{
			duration 120
			; Kill zone files are numbered by frame - at 20fps, this is twice the tick.
			; We add the zones in just before the kill time.
			kill_zone 1 roman_town_wall_small_damage_kill_zone_40.cas
			kill_zone 1 roman_town_wall_small_damage_kill_zone_50.cas
			kill_zone 1 roman_town_wall_small_damage_kill_zone_55.cas
			kill_zone 1 roman_town_wall_small_damage_kill_zone_70.cas
			; Physical switch should come after the last kill zone
			physical_switch 40
			item chiv_castle_wall_small_damadge
			{
				0 enable
				120 disable
			;}
			;item roman_town_wall_small_damage_anim
			;{
			;	0 enable
			;	0 anim 50 0 5
			;	0 dust_anim 30 0.6 2 2 0 3
			;	70 fade 50 1 0
			;	120 disable
			}
			item chiv_castle_wall_small_damadge_remains
			{
				0 enable
				120 disable
			}
		}
		level
